110template<class T>
111void MutablePriorityQueue<T>::heapifyUp(unsigned i) {
112 auto x = H[i];
113 while (i > 1 && *x < *H[parent(i)]) {
114 set(i, H[parent(i)]);
115 i = parent(i);
116 }
117 set(i, x);
118}

120template<class T>
121void MutablePriorityQueue<T>::heapifyDown(unsigned i) {
122 auto x = H[i];
123 while (true) {
124 unsigned k = leftChild(i);
125 if (k >= H.size())
126 break;
127 if (k + 1 < H.size() && *H[k + 1] < *H[k])
128 ++k; // right child of i
129 if (!(*H[k] < *x))
130 break;
131 set(i, H[k]);
132 i = k;
133 }
134 set(i, x);
135}
